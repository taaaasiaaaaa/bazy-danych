Zadanie 2
SELECT NAZWISKO, TRANSLATE(NAZWISKO, 'KLM', 'XXX') AS WOJNA_LITEROM
FROM PRACOWNICY;

Zadanie 3
SELECT NAZWISKO
FROM PRACOWNICY
WHERE INSTR(NAZWISKO,'L',1,1) BETWEEN 1 AND LENGTH(NAZWISKO)/2;

Zadanie 4
SELECT NAZWISKO, ROUND(PLACA_POD + PLACA_POD*0.15) AS PODWYZKA
FROM PRACOWNICY;

Zadanie 5
SELECT NAZWISKO, PLACA_POD,
PLACA_POD*0.2 AS INWESTYCJA, 
PLACA_POD*0.2*POWER(1+(10/100),10) AS KAPITAL,
PLACA_POD*0.2*POWER(1+(10/100),10) - PLACA_POD*0.2 AS ZYSK
FROM PRACOWNICY;

Zadanie 6
SELECT NAZWISKO, ZATRUDNIONY 
ROUND((DATE-'2000-01-01'-ZATRUDNIONY)/365) AS STAZ_W_2000
FROM PRACOWNICY
WHERE ZATRUDNIONY < DATE '2000-01-01';

Zadanie 7
SELECT NAZWISKO,
TO CHAR (ZATRUDNIONY,'MONTH, DD, YYYY') AS DATA_ZATRUDNIENIA
FROM PRACOWNICY
WHERE ID_ZESP = 20;

Zadanie 8
SELECT TO CHAR(CURRENT_DATE, 'DAY) AS DZIS
FROM PRACOWNICY
WHERE ROWNUM = 1;

Zadanie 9
SELECT NAZWA, ADRES,
CASE ADRES
	WHEN 'PIOTROWO 3A' THEN 'NOWE MIASTO'
	WHEN 'STRZELECKA 14 THEN' 'STARE MIASTO'
	WHEN 'MIELEZYNSKIEGO 30' THEN 'STARE MIASTO'
	WHEN 'WLODKOWICA 16' THEN 'GRUNWALD'
	ELSE 'BRAK'
	END AS DZIELNICA
FROM ZESPOLY;

Zadanie 10
SELECT NAZWISKO, PLACA_POD
CASE
	WHEN PLACA_POD < 480 THEN 'PONIZEJ 480'
	WHEN PLACA_POD > 480 THEN 'POWYZEJ 480'
	ELSE 'DOKLADNIE 480'
	END AS PROG
FROM PRACOWNICY;
